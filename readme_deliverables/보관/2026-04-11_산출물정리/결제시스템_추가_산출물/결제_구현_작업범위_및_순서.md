## 🔑 핵심 판단 — JWT 없이 뭘 할 수 있나?

```
JWT 없이 가능 ✅          JWT 있어야 가능 ⏳
────────────────────────────────────────────────────
Entity 작성               Controller에서 @AuthenticationPrincipal 사용
Repository 작성           실제 API 엔드포인트 인증 테스트
DTO 작성                  로그인 후 토큰 붙여서 Postman 테스트
PaymentGateway 작성
PaymentService 작성       ← 내부에서 memberId를 Long으로만 다루면 됨
Controller 골격 작성      ← @AuthenticationPrincipal 부분만 주석 처리
```

**결론: Entity → Repository → DTO → Gateway → Service 까지는 오늘 전부 완성 가능**
Controller는 골격만 만들고, `@AuthenticationPrincipal Member member` 부분은
hwan의 JWT가 완성되면 그때 연결하면 됩니다.

```java
package com.bookstore.shop.readme.domain;

import jakarta.persistence.*;
import lombok.*;

// @Entity : JPA가 이 클래스를 DB 테이블과 매핑하도록 선언
@Entity
// @Table : 매핑할 테이블명 명시 (기본값은 클래스명 소문자)
@Table(name = "cart")
@Getter @Setter
@NoArgsConstructor   // JPA는 기본 생성자가 반드시 필요
@AllArgsConstructor  // @Builder와 함께 사용
@Builder             // Cart.builder().member(member).build() 패턴 사용 가능
public class Cart extends BaseEntity {
    // BaseEntity로부터 상속: id(PK), createdAt, updatedAt 자동 관리

    // @OneToOne : Cart와 Member의 1:1 관계 선언
    // fetch = FetchType.LAZY : Cart를 조회할 때 Member는 즉시 로딩하지 않음
    //                         → N+1 문제 방지를 위해 LAZY 필수
    @OneToOne(fetch = FetchType.LAZY)
    // @JoinColumn : FK 컬럼명을 member_id로 지정
    // unique = true : 한 회원당 장바구니 1개만 허용 (DB 레벨 보장)
    @JoinColumn(name = "member_id", nullable = false, unique = true)
    private Member member;
}
```

```java
package com.bookstore.shop.readme.domain;

import jakarta.persistence.*;
import lombok.*;

@Entity
@Table(name = "cart_item")
@Getter @Setter
@NoArgsConstructor
@AllArgsConstructor
@Builder
public class CartItem extends BaseEntity {

    // @ManyToOne : 여러 CartItem이 하나의 Cart에 속함 (N:1)
    // LAZY : Cart 없이 CartItem만 조회할 때 Cart 쿼리 방지
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "cart_id", nullable = false)
    private Cart cart;

    // 어떤 상품을 담았는지 — N:1 (여러 CartItem이 같은 Product를 참조 가능)
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "product_id", nullable = false)
    private Product product;

    // @Builder.Default : @Builder 사용 시 기본값 1 보장
    //                    없으면 빌더로 생성 시 quantity = 0으로 초기화됨
    @Column(nullable = false)
    @Builder.Default
    private Integer quantity = 1;  // 담은 수량 (기본값: 1)

    // 장바구니 화면에서 체크박스 상태 저장 (체크된 항목만 결제로 넘어감)
    @Column(name = "is_checked", nullable = false)
    @Builder.Default
    private Boolean isChecked = true;  // 기본값: 체크 상태
}
```

#### `Order.java`

> ⚠️ `order`는 PostgreSQL 예약어이므로 테이블명을 반드시 `"order"` 로 쌍따옴표 처리해야 합니다.
> `@Table(name = "\"order\"")` — 쌍따옴표 없으면 PostgreSQL이 SQL 문법 오류 발생시킴.

```java
package com.bookstore.shop.readme.domain;

import jakarta.persistence.*;
import lombok.*;
import org.hibernate.annotations.ColumnDefault;
import org.springframework.data.annotation.CreatedDate;
import java.time.LocalDateTime;

@Entity
// ⚠️ PostgreSQL 예약어 회피: "order" 쌍따옴표 필수
// 쌍따옴표 없으면: ERROR: syntax error at or near "order"
@Table(name = "\"order\"")
@Getter @Setter
@NoArgsConstructor
@AllArgsConstructor
@Builder
public class Order extends BaseEntity {

    // 주문한 회원 — 여러 주문이 한 명의 회원에 속함 (N:1)
    // LAZY : 주문 목록 조회 시 불필요한 Member 쿼리 방지
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "member_id", nullable = false)
    private Member member;

    // 주문번호 — 예: "ORD-20260330-000001"
    // unique = true : 같은 주문번호 중복 방지 (DB 레벨)
    @Column(nullable = false, length = 30, unique = true)
    private String number;

    // @Enumerated(EnumType.STRING) : ENUM 값을 숫자(ORDINAL)가 아닌 문자열로 DB 저장
    //   → ORDINAL 사용 시 ENUM 순서 변경 시 데이터 깨짐 위험 → STRING 권장
    @Enumerated(EnumType.STRING)
    @Column(name = "order_status", nullable = false)
    // @Builder.Default : 빌더 사용 시 기본값 PENDING 보장
    @Builder.Default
    private OrderStatus orderStatus = OrderStatus.PENDING;

    // 총 상품 금액 (할인 적용 전)
    @Column(name = "total_price", nullable = false)
    @Builder.Default
    private Integer totalPrice = 0;

    // 할인 금액 (쿠폰, 포인트 등)
    @Column(name = "discount_amount", nullable = false)
    @Builder.Default
    private Integer discountAmount = 0;

    // 실제 결제 금액 = totalPrice - discountAmount
    // ⚠️ PaymentService에서 클라이언트가 보낸 amount와 이 값을 반드시 비교 검증!
    @Column(name = "final_price", nullable = false)
    @Builder.Default
    private Integer finalPrice = 0;

    // 배송받을 사람 정보 — 주문자와 다를 수 있음 (선물 등)
    @Column(name = "receiver_name", nullable = false, length = 50)
    private String receiverName;

    @Column(name = "receiver_phone", nullable = false, length = 20)
    private String receiverPhone;

    // 배송지 주소 (도로명 또는 지번)
    @Column(name = "delivery_address", nullable = false, length = 255)
    private String deliveryAddress;

    // 상세주소 (동/호수 등) — 선택 입력 → nullable
    @Column(name = "delivery_address_detail", length = 255)
    private String deliveryAddressDetail;

    // 우편번호
    @Column(name = "delivery_zip_code", nullable = false, length = 10)
    private String deliveryZipCode;

    // 배송 메모 — nullable (없을 수 있음)
    @Column(name = "delivery_memo", length = 300)
    private String deliveryMemo;

    // 주문 일시 — @CreatedDate로 자동 세팅 (BaseEntity의 createdAt과 별개)
    @CreatedDate
    @Column(name = "ordered_at", nullable = false, updatable = false)
    private LocalDateTime orderAt;

    // 취소 일시 — 주문 취소 시 이 필드에 현재 시간 저장
    @Column(name = "cancelled_at")
    private LocalDateTime cancelledAt;
}
```

#### `Payment.java`

```java
package com.bookstore.shop.readme.domain;

import jakarta.persistence.*;
import lombok.*;
import java.time.LocalDateTime;

@Entity
@Table(name = "payment")
@Getter @Setter
@NoArgsConstructor
@AllArgsConstructor
@Builder
public class Payment extends BaseEntity {

    // 어떤 주문의 결제인지 — 1:1 관계
    // unique = true : 한 주문당 결제 레코드 1개만 허용
    @OneToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "order_id", nullable = false, unique = true)
    private Order order;

    // 결제 수단명 — 예: "카드", "간편결제"
    // PG사에서 응답한 값 그대로 저장 (nullable — 결제 완료 전에는 모름)
    @Column(length = 30)
    private String method;

    // 어떤 PG사를 사용했는지
    @Enumerated(EnumType.STRING)
    @Column(length = 20)
    private PaymentProvider provider;

    // 결제 진행 상태 — READY → PAID / FAILED / CANCELLED
    @Enumerated(EnumType.STRING)
    @Column(name = "payment_status", length = 20, nullable = false)
    @Builder.Default
    private PaymentStatus paymentStatus = PaymentStatus.READY;

    // 결제 금액 — Order.finalPrice와 일치해야 함 (서비스에서 검증)
    @Column(nullable = false)
    @Builder.Default
    private Integer amount = 0;

    // 카카오/네이버 거래 ID — approve(취소) API 호출 시 필요
    // 토스는 이 필드 대신 paymentKey 사용
    @Column(name = "pg_tid", length = 200)
    private String pgTid;

    // 토스 paymentKey — 토스 취소 API URL에 path variable로 사용
    // 예: POST /v1/payments/{paymentKey}/cancel
    @Column(name = "payment_key", length = 200)
    private String paymentKey;

    // 할부 개월 수 — int 대신 Integer 사용: DB nullable 컬럼이므로 null 허용 필요
    // ⚠️ int(primitive)는 null 불가 → null이 오면 NullPointerException 발생
    @Column(name = "installment_months")
    private Integer installmentMonths;

    // 결제 완료 일시 — PaymentStatus가 PAID로 변경될 때 LocalDateTime.now()로 저장
    @Column(name = "paid_at")
    private LocalDateTime paidAt;

    // 취소 사유 — cancelPayment() 호출 시 입력받은 값 저장
    @Column(name = "cancel_reason", length = 300)
    private String cancelReason;
```

```java
package com.bookstore.shop.readme.controller;

import com.bookstore.shop.readme.dto.request.*;
import com.bookstore.shop.readme.dto.response.*;
import com.bookstore.shop.readme.service.PaymentService;
import lombok.RequiredArgsConstructor;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

// @RestController : @Controller + @ResponseBody 합친 어노테이션
//                   모든 메서드 반환값을 자동으로 JSON 직렬화하여 응답
@RestController
// @RequiredArgsConstructor : final 필드에 대한 생성자 자동 생성 → DI 간소화
@RequiredArgsConstructor
// "/order/payment" prefix를 모든 엔드포인트에 적용
@RequestMapping("/order/payment")
public class MyPaymentApiController {

    private final PaymentService paymentService;

    // ── JWT 임시 처리 ────────────────────────────────────────────────────
    // ✅ hwan JWT 완성 후 이 메서드 삭제 + 각 메서드 파라미터에 아래 추가:
    //    @AuthenticationPrincipal Member member
    //    Long memberId = member.getId();
    private Long getCurrentMemberId() {
        return 1L; // 임시값 — JWT 연동 전까지만 사용
    }

    // POST /order/payment/ready — 카카오/네이버 결제 준비
    // 프론트에서 provider, orderId, redirectUrl 등을 전송
    // 응답: PG사가 발급한 redirectUrl 또는 tid
    @PostMapping("/ready")
    public ResponseEntity<PaymentReadyResponse> readyPayment(
            @RequestBody PaymentReadyRequest request) {
            // TODO: hwan JWT 완성 후 → @AuthenticationPrincipal Member member 추가
            // Long memberId = member.getId();
            Long memberId = getCurrentMemberId();
        PaymentReadyResponse response = paymentService.requestReady(request, memberId);
        return ResponseEntity.ok(response);
    }

    // POST /order/payment/confirm — 토스 결제 최종 승인
    // 토스 successUrl 리다이렉트 이후 프론트에서 paymentKey, orderId, amount 전송
    @PostMapping("/confirm")
    public ResponseEntity<String> confirmPayment(
            @RequestBody PaymentConfirmRequest request) {
            // TODO: hwan JWT 완성 후 → @AuthenticationPrincipal Member member 추가
            Long memberId = getCurrentMemberId();
        paymentService.confirmToss(request, memberId);
        return ResponseEntity.ok("결제가 완료되었습니다.");
    }

    // POST /order/payment/approve — 카카오/네이버 결제 최종 승인
    // approval_url(카카오) / returnUrl(네이버) 이후 pg_token / paymentId 전송
    @PostMapping("/approve")
    public ResponseEntity<String> approvePayment(
            @RequestBody PaymentApproveRequest request) {
            // TODO: hwan JWT 완성 후 → @AuthenticationPrincipal Member member 추가
            Long memberId = getCurrentMemberId();
        paymentService.approvePayment(request, memberId);
        return ResponseEntity.ok("결제가 완료되었습니다.");
    }

    // POST /order/payment/fail — 토스 결제 실패 처리
    // failUrl 리다이렉트 이후 code, message 전송
    // ⚠️ SecurityConfig에서 이 URL만 permitAll() 처리 필요 (비인증 상태에서도 호출)
    @PostMapping("/fail")
    public ResponseEntity<String> failPayment(@RequestBody PaymentFailRequest request) {
        paymentService.failPayment(request);
        return ResponseEntity.ok("결제에 실패했습니다.");
    }
}
```
